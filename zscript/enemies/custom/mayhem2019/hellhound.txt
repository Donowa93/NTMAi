class NTM_Hellhound : NTM_Enemy {
	Default {
		Health 140;
		Radius 24;
		Height 56;
		Speed 25;
		PainChance 50;
		Mass 300;
		Obituary "%o was brutally mauled by a Hellhound.";
		Tag "Hellhound";
		SeeSound "NTM_Hellhound/sight";
		ActiveSound "NTM_Hellhound/active";
		PainSound "NTM_Hellhound/pain";
		DeathSound "NTM_Hellhound/death";
		MaxDropOffHeight 80;
		NTM_Enemy.TicsPerChase 3;
		NTM_Enemy.LeapDistance 360;
	}
	
	States {
		Spawn:
			NDOG A 10 A_Look;
			Loop;
		
		DukeSee:
			NDOG ABCDEF 3 A_NTM_DukeChase;
			Loop;
		
		BloodSee:
			NDOG ABCDEF 3 A_NTM_BloodChase;
			Loop;
		
		SWSee:
			NDOG A 1 A_NTM_SWChase;
			NDOG AA 1 A_NTM_SWChase2;
			NDOG B 1 A_NTM_SWChase;
			NDOG BB 1 A_NTM_SWChase2;
			NDOG C 1 A_NTM_SWChase;
			NDOG CC 1 A_NTM_SWChase2;
			NDOG D 1 A_NTM_SWChase;
			NDOG DD 1 A_NTM_SWChase2;
			NDOG E 1 A_NTM_SWChase;
			NDOG EE 1 A_NTM_SWChase2;
			NDOG F 1 A_NTM_SWChase;
			NDOG FF 1 A_NTM_SWChase2;
			Loop;
		
		Melee:
			NDOG G 0 A_StartSound("NTM_Hellhound/melee");
			NDOG GH 3 A_FaceTarget;
			NDOG I 8 A_CustomMeleeAttack(25);
			Goto See;
		
		Pain:
			NDOG J 10 A_Pain;
			Goto See;
		
		Death:
			NDOG K 5 A_NTM_Death;
			NDOG L 5 A_NoBlocking;
			NDOG MNO 5;
			NDOG Q -1;
			Stop;
		
		Raise:
			NDOG ONMLK 5;
			Goto See;
			
		Leap:
			NDOG C 1 A_NTM_Leap;
			Loop;
	}
}

class NTM_ShadowHellhound : NTM_Hellhound {
	Default {
		RenderStyle "Stencil";
		Alpha 0.3;
		
		+Shadow
	}
}
